● Seleziona tutti gli ospiti che sono nati dopo il 1988

----- OPTION 1 -----

SELECT `name`, `lastname`, `date_of_birth`
FROM `ospiti`
WHERE date_of_birth > '1988-12-31';

----- OPTION 2 -----

SELECT `name`, `lastname`, `date_of_birth`
FROM `ospiti`
WHERE date_of_birth >= '1989-01-01';

----- OPTION 3 -----

SELECT `name`, `lastname`, `date_of_birth`
FROM `ospiti`
WHERE YEAR(date_of_birth) > 1988;
